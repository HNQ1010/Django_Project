    Filled Area Plot With Plotly Express

  • px.area creates a stacked area plot. Each filled area corresponds to one value of the column given by the line_group parameter.
    •             df = px.data.gapminder()
            fig = px.area(df, x = 'year', y = 'pop', color = 'continent', line_group = 'country')
            fig.show()

    Fill Area Plot in Dash

                import dash
            from dash import dcc
            from dash import html
            from dash import Input, Output
            df = px.data.gapminder()
            countries = df.country.drop_duplicated().sample(n=10, random_state=42)
            df = df[df.country.isin(countries)]
            app = dash.Dash(__name__)
            app.layout = html.Div([html.P('Select y-axis'), dcc.Dropdown(id = 'select_list', value = 'pop',
                                                            options = [{'label':'x', 'value':x} for x in ['lifeExp', 'pop', 'gdpPercap']]
                                    dcc.Graph(id = 'area-chart')
            @app.callback(Output('area-chart', 'figure'), [Input('select_list', 'value'])
            def create_area_chart(y):
                fig = px.area(df, x = 'year', y = y, line_group = 'country', color = 'continent')
                return fig
            app.run_server(debug = True)

    Stacked Area Chart

  • The stackgroup parameter is used to add the y values of the different traces in the same group. Traces in the same group fill up
    to the next trace of the group.
    •             x = ['Winter', 'Spring', 'Summer', 'Fall']
            fig = go.Figure()
            fig.add_trace(go.Scatter(x = x, y = [40, 60, 40, 10], hoverinfo = 'x + y', mode = 'lines',
                                    line = {'width': 0.5, 'color':'rgb(131, 90, 241)'},
                                    stackgroup = 'one'))
            fig.add_trace(go.Scatter(x = x, y = [20, 10, 10, 60], hoverinfo = 'x + y', mode = 'lines',
                                    line = {'width': 0.5, 'color':'rgb(111, 231, 219)'},
                                    stackgroup = 'one'))
            fig.add_trace(go.Scatter(x = x, y = [40, 30, 50, 30], hoverinfo = 'x + y', mode = 'lines',
                                    line = {'width':0.5, 'color':'rgb(184, 247, 212)'},
                                    stackgroup = 'one'))
            fig.update_layout(yaxis_range = [0, 100])
            fig.show()

    Stacked Area Chart With Normalized Values

                x = ['Winter', 'Spring', 'Summer', 'Fall']
            fig = go.Figure()
            fig.add_trace(go.Scatter(x = x, y = [40, 20, 30, 40], mode = 'lines', line = {'width':0.5, 'color':'rgb(184, 247, 212)'},
                                        stackgroup = 'one', groupnorm = 'percent'))
            fig.add_trace(go.Scatter(x = x, y = [50, 70, 40, 60], mode = 'lines', line = {'width': 0.5, 'color': 'rgb(111, 231, 219)'},
                                        stackgroup = 'one', groupnorm = 'percent'))
            fig.add_trace(go.Scatter(x = x, y = [70, 80, 60, 70], mode = 'lines', line = {'width':0.5, 'color': 'rgb(127, 166, 238)'},
                                        stackgroup = 'one', groupnorm = 'percent'))
            fig.add_trace(go.Scatter(x =x, y = [100, 100, 100, 100], mode= 'lines', line = {'width':0.5, 'color': 'rgb(131, 90, 241)'},
                                        stackgroup = 'one', groupnorm = 'percent'))
            fig.update_layout(showlegend = True, xaxis_type = 'category', yaxis = {'type': 'linear', 'range':[0, 100], 'ticksuffix': '%'})
            fig.show()
